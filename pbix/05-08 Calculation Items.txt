)

PY YTD = 
CALCULATE(
    SELECTEDMEASURE(),
    SAMEPERIODLASTYEAR('Date'[Date]),
    'Time Intelligence'[Time Calculation] = "YTD"
)

    SELECTEDMEASURE(),
    'Time Intelligence'[Time Calculation] = "PY"
)

DIVIDE(
    CALCULATE(
        SELECTEDMEASURE(),
        'Time Intelligence'[Time Calculation]="YOY"

    CALCULATE(
        SELECTEDMEASURE(),

    )
